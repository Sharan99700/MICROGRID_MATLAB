        function updateWeightSum() {
            try {
                const w1 = parseFloat(document.getElementById('w_tri')?.value) || 0;
                const w2 = parseFloat(document.getElementById('w_ori')?.value) || 0;
                const w3 = parseFloat(document.getElementById('w_lsi')?.value) || 0;
                const w4 = parseFloat(document.getElementById('w_esi')?.value) || 0;
                const w5 = parseFloat(document.getElementById('w_ecsi')?.value) || 0;
                const sum = (w1 + w2 + w3 + w4 + w5).toFixed(2);
                const elem = document.getElementById('weight_sum');
                if (elem) {
                    elem.textContent = `Sum: ${sum}`;
                    elem.style.color = Math.abs(sum - 1.0) < 0.01 ? '#38ef7d' : '#ff6b6b';
                }
                updatePillarPercentages(w1, w2, w3, w4, w5);
            } catch (e) { console.error('updateWeightSum:', e); }
        }

        function updatePillarPercentages(w_tri, w_ori, w_lsi, w_esi, w_ecsi) {
            const ids = { 'tri-weight': w_tri, 'ori-weight': w_ori, 'lsi-weight': w_lsi,
                          'esi-weight': w_esi, 'ecsi-weight': w_ecsi };
            Object.entries(ids).forEach(([id, w]) => {
                const el = document.getElementById(id);
                if (el) el.textContent = (w * 100).toFixed(0);
            });
            const fIds = { 'formula-tri': w_tri, 'formula-ori': w_ori, 'formula-lsi': w_lsi,
                           'formula-esi': w_esi, 'formula-ecsi': w_ecsi };
            Object.entries(fIds).forEach(([id, w]) => {
                const el = document.getElementById(id);
                if (el) el.textContent = w.toFixed(2);
            });
        }

        function findLocationInDatabase(searchTerm) {
            searchTerm = searchTerm.toLowerCase().replace(/[^a-z0-9]/g, '');
            for (const state in locationDatabase) {
                for (const key in locationDatabase[state]) {
                    const loc = locationDatabase[state][key];
                    const name = loc.name.toLowerCase().replace(/[^a-z0-9]/g, '');
                    if (name === searchTerm || key === searchTerm ||
                        name.includes(searchTerm) || searchTerm.includes(name)) {
                        return { state, key, ...loc };
                    }
                }
            }
            return null;
        }

        function submitAndCalculate() {
            const fileInput = document.getElementById('file_input');
            if (!fileInput?.files?.length) {
                alert('Please select a file first.');
                return;
            }
            const file = fileInput.files[0];
            const ext  = file.name.split('.').pop().toLowerCase();

            showLoading('Parsing file...');

            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    let rows;
                    if (ext === 'csv') {
                        const parsed = Papa.parse(e.target.result, { header: true, dynamicTyping: true, skipEmptyLines: true });
                        rows = parsed.data;
                    } else {
                        const wb = XLSX.read(e.target.result, { type: 'array' });
                        rows = XLSX.utils.sheet_to_json(wb.Sheets[wb.SheetNames[0]]);
                    }
                    processUploadedRows(rows, file.name);
                } catch(err) {
                    hideLoading();
                    alert('Error parsing file: ' + err.message);
                }
            };
            if (ext === 'csv') reader.readAsText(file);
            else reader.readAsArrayBuffer(file);
        }

        function processUploadedRows(rows, filename) {
            try {
                // ── Detect columns (case-insensitive) ──
                if (!rows || rows.length === 0) { alert('File is empty.'); hideLoading(); return; }
                const sample = rows[0];
                const keys   = Object.keys(sample);

                // ── Robust column finder: try exact match first, then partial ──
                const findCol = (...names) => {
                    const norm = s => s.toLowerCase().replace(/[^a-z0-9]/g,'');
                    // 1st pass: exact normalised match
                    for (const n of names) {
                        const hit = keys.find(k => norm(k) === norm(n));
                        if (hit) return hit;
                    }
                    // 2nd pass: starts-with
                    for (const n of names) {
                        const hit = keys.find(k => norm(k).startsWith(norm(n)));
                        if (hit) return hit;
                    }
                    // 3rd pass: includes (most permissive — last resort)
                    for (const n of names) {
                        const hit = keys.find(k => norm(k).includes(norm(n)));
                        if (hit) return hit;
                    }
                    return undefined;
                };

                // Priority order matches the backend's fuzzy column detection
                const colSolar = findCol('GHI_Wh_m2','GHI_W_per_m2','GHI_kWh_m2','GHI','ghi',
                                          'SolarGHI','solar_irradiance','Irradiance','Solar','solar');
                const colWind  = findCol('WindSpeed_10m_mps','WindSpeed_mps','WindSpeed','Wind_Speed',
                                          'wind_mps','wind_ms','Wind','wind','wind_speed');
                const colLoad  = findCol('Load_kW','load_kW','Load_KW','Demand_kW','Power_kW',
                                          'Load','load','Demand','Power','Consumption');

                if (!colSolar || !colWind || !colLoad) {
                    alert('Required columns not found.\nExpected: GHI_Wh_m2, WindSpeed_10m_mps, Load_kW\nFound: ' + keys.join(', '));
                    hideLoading(); return;
                }
                console.log('Column mapping → GHI:', colSolar, '| Wind:', colWind, '| Load:', colLoad);

                const N = rows.length;
                const solar = rows.map(r => parseFloat(r[colSolar]) || 0);
                const wind  = rows.map(r => parseFloat(r[colWind])  || 0);
                const load  = rows.map(r => parseFloat(r[colLoad])  || 0);

                // ── Unit auto-detection (mirrors mpbsi_backend.py exactly) ──
                // GHI: if peak ≤ 10 → data is in kW/m² or kWh/m² → keep as kW/m² for display
                //      if peak >  10 → data is in Wh/m²            → convert to kW/m² (/1000)
                const solarPeak = Math.max(...solar.filter(v => isFinite(v)));
                const solarIsWh = solarPeak > 10;           // true  → Wh/m²  → divide by 1000
                const avgSolar  = solar.reduce((a,b)=>a+b,0) / N / (solarIsWh ? 1000 : 1);
                // avgSolar is now always in kW/m²

                // Load: column is expected in kW — no conversion needed
                // Backend auto-detects but does NOT scale load values
                const avgLoad  = load.reduce((a,b)=>a+b,0) / N;  // kW
                const peakLoad = Math.max(...load.filter(v => isFinite(v)));

                const avgWind  = wind.reduce((a,b)=>a+b,0) / N;

                // Annual energy: avgLoad(kW) × N(hours) / 1000 = MWh
                const annual   = avgLoad * N / 1000; // MWh

                uploadedData = {
                    solar, wind, load,
                    avg_load: +avgLoad.toFixed(4), peak_load: +peakLoad.toFixed(4),
                    avg_ghi_kw: +avgSolar.toFixed(6), avg_wind: +avgWind.toFixed(4),
                    complete: true
                };

                // ── Update legacy display IDs (Optimize tab still uses these) ──
                const setEl = (id, val) => { const el = document.getElementById(id); if(el) el.textContent = val; };
                const setVal = (id, val) => { const el = document.getElementById(id); if(el) el.value = val; };
                setEl('calc_avg_solar',   avgSolar.toFixed(4));
                setEl('calc_avg_wind',    avgWind.toFixed(2));
                setEl('calc_peak_load',   peakLoad.toFixed(2));
                setEl('calc_peak_load_2', peakLoad.toFixed(2));
                setEl('calc_avg_load',    avgLoad.toFixed(2));
                setEl('calc_avg_load_2',  avgLoad.toFixed(2));
                setEl('calc_annual',      annual.toFixed(2));
                setVal('peak_load_input', peakLoad.toFixed(2));
                setVal('avg_load_input',  avgLoad.toFixed(2));
                setVal('annual_demand',   annual.toFixed(2));

                // ── Populate Summary tab ──
                setEl('summary_filename', filename);
                setEl('summary_rowcount', N + ' hourly rows');
                setEl('sum_avg_solar',  avgSolar.toFixed(4));
                setEl('sum_avg_wind',   avgWind.toFixed(2));
                setEl('sum_peak_load',  peakLoad.toFixed(2));
                setEl('sum_avg_load',   avgLoad.toFixed(2));
                setEl('sum_annual',     annual.toFixed(2));
                setEl('sum_rows',       N.toLocaleString());

                // ── Build monthly breakdown chart ──
                try {
                    const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
                    const mDays  = [31,28,29,31,30,31,30,31,31,30,31,30]; // approx hours per month
                    const mHours = mDays.map(d => d*24);
                    let mSolar=[], mWind=[], mLoad=[];
                    let offset=0;
                    for(let m=0;m<12;m++){
                        const h = Math.min(mHours[m], N-offset);
                        if(h<=0){mSolar.push(0);mWind.push(0);mLoad.push(0);}
                        else{
                            const sl = solar.slice(offset,offset+h);
                            const wl = wind.slice(offset,offset+h);
                            const ll = load.slice(offset,offset+h);
                            mSolar.push((sl.reduce((a,b)=>a+b,0)/h/(solarIsWh ? 1000 : 1)));
                            mWind.push(wl.reduce((a,b)=>a+b,0)/h);
                            mLoad.push(ll.reduce((a,b)=>a+b,0)/h);
                        }
                        offset += mHours[m];
                    }
                    if(typeof Plotly !== 'undefined'){
                        const dark = { paper_bgcolor:'rgba(0,0,0,0)', plot_bgcolor:'rgba(10,12,35,0.6)',
                                       font:{color:'#c8d0e8',family:'Space Grotesk'},
                                       xaxis:{color:'#8b9dc3',gridcolor:'rgba(102,126,234,0.1)'},
                                       yaxis:{color:'#8b9dc3',gridcolor:'rgba(102,126,234,0.12)',title:'kW or kW/m²'},
                                       yaxis2:{overlaying:'y',side:'right',color:'#8b9dc3',
                                               gridcolor:'rgba(0,0,0,0)',title:'m/s'},
                                       legend:{bgcolor:'rgba(10,12,35,0.7)',bordercolor:'rgba(102,126,234,0.3)',borderwidth:1},
                                       margin:{l:50,r:60,t:20,b:40}, barmode:'group' };
                        Plotly.newPlot('summary_monthly_chart',[
                            {name:'☀️ Avg Solar (kW/m²)', x:months, y:mSolar.map(v=>+v.toFixed(4)),
                             type:'bar', marker:{color:'rgba(255,215,0,0.75)'}},
                            {name:'📊 Avg Load (kW)', x:months, y:mLoad.map(v=>+v.toFixed(2)),
                             type:'bar', marker:{color:'rgba(102,126,234,0.75)'}},
                            {name:'🌬️ Avg Wind (m/s)', x:months, y:mWind.map(v=>+v.toFixed(2)),
                             type:'scatter', mode:'lines+markers', yaxis:'y2',
                             line:{color:'#00BFFF',width:2.5},
                             marker:{color:'#00BFFF',size:6}},
                        ], dark, {responsive:true});
                    }
                } catch(e){ console.warn('Monthly chart error:', e); }

                // ── Show summary content, hide empty state ──
                document.getElementById('summary_empty_state').style.display = 'none';
                document.getElementById('summary_content').style.display = 'block';

                // ── Show/init profile viewer cards ──
                document.getElementById('solar_profile_card').style.display = 'block';
                document.getElementById('wind_profile_card').style.display  = 'block';
                updateSolarProfile();
                updateWindProfile();

                // ── Auto-switch to Summary tab ──
                switchTab('initial_sizing');

                document.getElementById('upload_status').innerHTML =
                    '<p style="color:#38ef7d;">✅ <strong>' + filename + '</strong> loaded — ' + N + ' rows. See the <strong>Summary tab</strong> for full statistics.</p>';

                hideLoading();

            } catch(err) {
                hideLoading();
                alert('Error processing file: ' + err.message);
                console.error(err);
            }
        }

        function updateSolarProfile() {
            if (!uploadedData.complete) return;
            const day   = parseInt(document.getElementById('solar_profile_day')?.value) || 1;
            const mode  = document.getElementById('solar_profile_mode')?.value || 'daily';
            const res   = document.getElementById('solar_profile_resolution')?.value || 'hourly';
            const solar = uploadedData.solar;
            if (!solar) return;

            let startH = (day - 1) * 24;
            let endH   = startH + (mode === 'daily' ? 24 : mode === 'weekly' ? 168 : 720);
            endH = Math.min(endH, solar.length);

            let rawData = solar.slice(startH, endH);
            let xLabels = Array.from({length: rawData.length}, (_, i) => 'H' + (startH + i + 1));

            if (res === '6hourly') {
                rawData = rawData.filter((_, i) => i % 6 === 0);
                xLabels = xLabels.filter((_, i) => i % 6 === 0);
            } else if (res === 'daily') {
                const days = [];
                for (let i = 0; i < rawData.length; i += 24) {
                    const slice = rawData.slice(i, i + 24);
                    days.push(slice.reduce((a,b)=>a+b,0) / slice.length);
                }
                rawData = days;
                xLabels = Array.from({length: days.length}, (_, i) => 'Day ' + (Math.floor(startH/24) + i + 1));
            }

            const avgVal = rawData.reduce((a,b)=>a+b,0)/rawData.length;
            const maxVal = Math.max(...rawData);
            // Apply same unit conversion as processUploadedRows: if raw values > 10 → Wh/m² → /1000 for kW/m²
            const profSolarIsWh = maxVal > 10;
            const profDiv = profSolarIsWh ? 1000 : 1;
            document.getElementById('solar_profile_stats').innerHTML =
                '<strong>☀️ Solar Stats:</strong> Avg: ' + (avgVal/profDiv).toFixed(4) + ' kW/m² | Peak: ' + (maxVal/profDiv).toFixed(4) + ' kW/m² | Hours: ' + rawData.length;

            Plotly.newPlot('solar_profile_chart', [{
                x: xLabels, y: rawData.map(v => v/profDiv),
                type: 'scatter', mode: 'lines',
                fill: 'tozeroy', fillcolor: 'rgba(255,215,0,0.15)',
                line: { color: '#FFD700', width: 2 }, name: 'GHI'
            }], {
                paper_bgcolor: 'rgba(0,0,0,0)', plot_bgcolor: 'rgba(0,0,0,0)',

                yaxis: { title: 'GHI (kW/m²)', color: '#fff', gridcolor: 'rgba(255,255,255,0.1)' },
                font: { color: '#fff' }
            }, { responsive: true, displayModeBar: false });
        }

        // ══════════════════════════════════════════════════════════════════
        //  SIDEBAR FILE INJECTION — called by Streamlit when file uploaded
        //  base64Data: base64-encoded file bytes; filename: original name
        // ══════════════════════════════════════════════════════════════════
        window.receiveFileFromSidebar = function(base64Data, filename) {
            try {
                const ext = filename.split('.').pop().toLowerCase();
                const byteStr = atob(base64Data);
                const arr = new Uint8Array(byteStr.length);
                for(let i=0;i<byteStr.length;i++) arr[i] = byteStr.charCodeAt(i);
                const blob = new Blob([arr], {
                    type: ext === 'csv'
                        ? 'text/csv'
                        : 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
                });
                const file = new File([blob], filename, {type: blob.type});

                // ── Inject into file input so the native uploader also shows the file ──
                const dt = new DataTransfer();
                dt.items.add(file);
                const fileInput = document.getElementById('file_input');
                if(fileInput) {
                    fileInput.files = dt.files;
                    onFileSelected(fileInput);
                }

                // ── Parse and process directly ──
                const reader = new FileReader();
                reader.onload = function(e) {
                    try {
                        let rows;
                        if(ext === 'csv'){
                            const parsed = Papa.parse(e.target.result, {header:true, dynamicTyping:true, skipEmptyLines:true});
                            rows = parsed.data;
                        } else {
                            const wb = XLSX.read(e.target.result, {type:'array'});

        function getFinancialParams() {
            return {
                cost_pv:      parseFloat(document.getElementById('cost_pv')?.value)       || 55000,
                cost_wind:    parseFloat(document.getElementById('cost_wind')?.value)      || 120000,
                cost_battery: parseFloat(document.getElementById('cost_battery')?.value)   || 22000,
                cost_fc:      parseFloat(document.getElementById('cost_fc')?.value)        || 110000,
                cost_elec:    parseFloat(document.getElementById('cost_elec')?.value)      || 70000,
                cost_h2:      parseFloat(document.getElementById('cost_h2')?.value)        || 80000,
                discount_rate: parseFloat(document.getElementById('fp_discount')?.value)   || 8,
                lifetime:      parseInt(document.getElementById('fp_lifetime')?.value)      || 20,
                om_rate:       parseFloat(document.getElementById('fp_om_rate')?.value)    || 2,
                bat_replace_yr: parseInt(document.getElementById('fp_bat_replace')?.value) || 10,
            };
        }

        function computeCapexNpc(bestX, rel) {
            if (!Array.isArray(bestX) || bestX.length < 6) return null;
            const fp = getFinancialParams();
            const [pv_kw, wt_kw, bat_kwh, el_kw, h2_kwh, fc_kw] = bestX.map(v => parseFloat(v) || 0);
            const r = fp.discount_rate / 100;
            const n = fp.lifetime;
            const om_rate = fp.om_rate / 100;

            const Cost_PV  = fp.cost_pv      * pv_kw;
            const Cost_WT  = fp.cost_wind     * wt_kw;
            const Cost_Bat = fp.cost_battery  * bat_kwh;
            const Cost_EL  = fp.cost_elec     * el_kw;
            const Cost_H2  = fp.cost_h2       * h2_kwh;
            const Cost_FC  = fp.cost_fc       * fc_kw;
            const CAPEX    = Cost_PV + Cost_WT + Cost_Bat + Cost_EL + Cost_H2 + Cost_FC;

            const fc_yr    = 10, el_yr = 15;
            const Bat_repl = Cost_Bat / Math.pow(1 + r, fp.bat_replace_yr);
            const FC_repl  = Cost_FC  / Math.pow(1 + r, fc_yr);
            const EL_repl  = Cost_EL  / Math.pow(1 + r, el_yr);
            const OM_ann   = om_rate * CAPEX;
            const annuity  = (1 - Math.pow(1 + r, -n)) / r;
            const OM_PV    = OM_ann * annuity;
            const NPC      = CAPEX + Bat_repl + FC_repl + EL_repl + OM_PV;

            const ann_load_kwh = (rel && rel.annual_load_MWh ? rel.annual_load_MWh * 1000 : 0);
            const LCOE = (ann_load_kwh > 0 && n > 0) ? NPC / (ann_load_kwh * n) : 0;

            return { CAPEX, NPC, LCOE, OM_ann, OM_PV,
                     Cost_PV, Cost_WT, Cost_Bat, Cost_EL, Cost_H2, Cost_FC,
                     Bat_repl, FC_repl, EL_repl, fp, r, n };
        }

            return '₹' + (v / 1e7).toFixed(3) + ' Cr';

        function pctFmt(v, total) {
            return total > 0 ? (v / total * 100).toFixed(1) + '%' : '—';
        }
